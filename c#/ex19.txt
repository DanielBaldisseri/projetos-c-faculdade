using System;

class Program
{
    static void Main()
    {
        Console.WriteLine("Digite um número:");
        int num = int.Parse(Console.ReadLine()!);

        int divisores = 0;

        for (int i = 1; i <= num; i++)
        {
            if (num % i == 0)
                divisores++;
        }

        if (divisores == 2)
            Console.WriteLine("É primo");
        else
            Console.WriteLine("Não é primo");
    }
}
