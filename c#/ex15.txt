using System;

class Program
{
    static void Main()
    {
        int pares = 0, impares = 0;

        for (int i = 1; i <= 10; i++)
        {
            Console.WriteLine($"Digite o número {i}:");
            int num = int.Parse(Console.ReadLine()!);

            if (num % 2 == 0)
                pares++;
            else
                impares++;
        }

        Console.WriteLine($"Pares: {pares}");
        Console.WriteLine($"Ímpares: {impares}");
    }
}
