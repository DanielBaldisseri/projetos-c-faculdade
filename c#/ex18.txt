using System;

class Program
{
    static void Main()
    {
        int soma = 0;

        for (int i = 1; i <= 5; i++)
        {
            Console.WriteLine($"Digite a idade {i}:");
            soma += int.Parse(Console.ReadLine()!);
        }

        double media = soma / 5.0;

        Console.WriteLine($"Média = {media:F2}");
    }
}
